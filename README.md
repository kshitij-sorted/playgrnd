1. sudo pacman -S git
2. sudo pacman -S github-cli
3. go to desired folder
4. git init
5. git status
6. git auto login
7. git remote add origin <repo_link>
8. git remote -v
9. git remote set-url origin <NEW_GIT_URL_HERE>
10. git add ./
11. git status
12. git commit - m ""
13. git push -u origin main
14. git pull origin main
